Quick start (Node.js)

// Sign a receipt via reference API

  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    // Use your API key when available (paid plans)
    // 'Authorization': 'Bearer YOUR_API_KEY'
  },
  body: JSON.stringify({ payload: { hello: 'world' } })
});
const receipt = await res.json();
console.log('receipt_id', receipt.receipt_id);
