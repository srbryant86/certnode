For Developers

Node.js

Sign and verify receipts programmatically

// install
npm i @certnode/sdk

// sign
const { sign } = require('@certnode/sdk')
const { receipt } = await sign({ payload: { hello: 'world' } })

// verify
const { ok } = await verifyReceipt({ receipt, jwks })
